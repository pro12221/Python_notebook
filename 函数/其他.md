## 递归调用
```python
# 直接调用
def f():
    print(111)
    f()
f()
# 间接调用
def f1():
    print(111)
    f2()
def f2():
    print(222)
    f1()
f2()

# 例子1
def age(n):
    if n==1:
        return 18
    res=age(n-1)+10
    return res
age(10)
# 例子2
l = [1, [2, [3, [4]]]]

def f(lst):  # 添加参数
    for item in lst:
        if type(item) is list:
            f(item)  # 递归调用时传递参数
        else:
            print(item)

f(l)  # 调用时传递参数

```
